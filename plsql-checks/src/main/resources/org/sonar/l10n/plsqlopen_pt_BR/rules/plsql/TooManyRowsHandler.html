Quando um comando SELECT INTO causa a exceção pré-definida TOO_MANY_ROWS os valores das variáveis na cláusula INTO ficam indefinidos.

Código em desconformidade

begin
  select empno
    into var
    from emp;
exception
  when too_many_rows then
    null;
end;

Código correto

begin
  select empno
    into var
    from emp;
exception
  when too_many_rows then
    var := null;
end;
